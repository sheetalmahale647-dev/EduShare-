<script>
const supabaseUrl = "https://cexjkuowkizjzmwjqnub.supabase.co/rest/v1/";
const supabaseKey = "sb_publishable_KJnVc7ywGt6qw5AokQ_AVA_EAX6YJZh";

const supabase = window.supabase.createClient(
    supabaseUrl,
    supabaseKey
);

async function loadNotes(){

    const container =
    document.getElementById("notesContainer");

    container.innerHTML = "Loading notes...";

   
    document.getElementById("filterClass").value;

    const searchText =
    document.getElementById("searchBox").value.toLowerCase();

    let query = supabase
    .from("notes")
    .select("*")
    .order("id", {ascending:false});

    if(selectedClass != "All"){
        query = query.eq("class", selectedClass);
    }

    const { data, error } = await query;

    if(error){
        container.innerHTML = "Error loading notes";
        console.log(error);
        return;
    }

    if(data.length === 0){
        container.innerHTML = "No notes found.";
        return;
    }

    container.innerHTML = "";

    data.forEach(note => {

if(
note.title.toLowerCase().includes(searchText) ||
note.subject.toLowerCase().includes(searchText)
){

container.innerHTML += `
<div class="note">
    <h3>${note.title}</h3>
    <p><b>Class:</b> ${note.class}</p>
    <p><b>Subject:</b> ${note.subject}</p>

    <button onclick="window.open('${note.file_url}','_blank')">
    Download
    </button>
</div>
`;

}

});

}

loadNotes();
</script>
